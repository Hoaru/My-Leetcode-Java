class Solution {
    public boolean isValid(String s) {
        Stack<Character> st = new Stack<>();
        for(int i = 0; i < s.length(); i++){
            char now = s.charAt(i);
            if(st.size() == 0 || now == '(' || now == '[' || now == '{'){
                st.push(now);
                continue;
            }
            char top = st.pop();
            if(now == ']'){
                if(top == '[')
                    continue;
                else return false;
            }
            else if(now == '}'){
                if(top == '{')
                    continue;
                else return false;
            }
            else if(now == ')'){
                if(top == '(')
                    continue;
                else return false;
            }
            else return false;
        }
        if(st.size() == 0) return true;
        else return false;
    }
}
