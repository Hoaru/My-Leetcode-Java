class Solution {
    public boolean validPalindrome(String s) {
        int i = 0; 
        int j = s.length() - 1;
        for(; i < j; i++, j--){
            if(s.charAt(i) == s.charAt(j)) continue;
            else{
                return helper(i + 1, j, s) || helper(i, j - 1, s);
            }
        }
        return true;
    }

    public boolean helper(int x, int y, String s){
        for(int i = x, j = y; i < j; i++, j--){
            if(s.charAt(i) == s.charAt(j)) continue;
            else return false;
        }
        return true;
    }
}
