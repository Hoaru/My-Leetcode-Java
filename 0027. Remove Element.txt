class Solution {
    public int removeElement(int[] nums, int val) {
        int n = nums.length;
        int res = 0;
        int i = 0, j = 0;
        for(; j < n;){
            if(nums[j] != val){
                nums[i] = nums[j];
                i++;
                j++;
            }
            else{
                res++;
                j++;
            }
        }
        return n - res;
    }
}
