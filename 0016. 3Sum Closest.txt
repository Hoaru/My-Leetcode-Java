class Solution {
    public static int threeSumClosest(int[] nums, int target) {
        int diff = Integer.MAX_VALUE;
        int sum = nums[0] + nums[1] + nums[2];
        Arrays.sort(nums);
        for(int i = 0; i < nums.length; i++){
            int l = i + 1;
            int r = nums.length - 1;
            while(l < r){
                int newSum = nums[i] + nums[l] + nums[r];
                int newDiff = Math.abs(target - newSum);
                if(newDiff == 0) return target;
                if(newDiff < diff){
                    diff = newDiff;
                    sum = newSum;
                }
                else if(newSum < target) l++;
                else if(newSum > target) r--;
            }
        }
        return sum;
    }
}
