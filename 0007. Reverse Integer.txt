class Solution {
    public int reverse(int x) {
        ArrayList<Integer> arr = new ArrayList<>();
        while(x != 0){
            arr.add(x % 10);
            x = x / 10;
        }
        long sum = 0;
        for(int i = 0; i < arr.size(); i++){
            sum = sum * 10 + arr.get(i);
        }
        if(sum > Integer.MAX_VALUE || sum < Integer.MIN_VALUE) return 0;
        return (int)sum;
    }
}
