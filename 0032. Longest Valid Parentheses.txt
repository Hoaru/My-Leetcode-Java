class Solution {
        public int longestValidParentheses(String s) {
            int max = 0;
            Stack<Integer> st = new Stack<>();
            for(int i = 0; i < s.length(); i++){
                if(s.charAt(i) == '('){
                    st.push(i);
                }
                else{
                    if(!st.isEmpty() && s.charAt(st.peek()) == '('){
                        st.pop();
                        if(st.isEmpty()){
                            max = Math.max(max, i + 1);
                        }
                        else{
                            max = Math.max(max, i - st.peek());
                        }
                    }
                    else{
                        st.push(i);
                    }
                }
            }
            return max;
        }
    }
